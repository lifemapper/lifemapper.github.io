353 # @return: An ElementTree element representing the object

356 def serialize(obj, parent=None):
357  value = None
358  attrib = {}
359  if hasattr(obj, 'value'):
360  value = obj.value
361  elif isinstance(obj, (StringType, UnicodeType)):
362  value = obj
363 
364  fltr = lambda x: not x.startswith('_') and not x == "attrib" and \
365  not x == "value"
366  objAttribs = [att for att in dir(obj) if fltr(att)]
367 
368  if hasattr(obj, 'attrib'):
369  for k, v in [(key, obj.attrib[key]) for key in obj.attrib.keys()]:
370  attrib[k] = v
371  try:
372  atts = obj.getAttributes()
373  for key in atts.keys():
374  if isinstance(atts[key], (IntType, StringType, FloatType, UnicodeType)):
375  attrib[key] = str(atts[key])
376  elif isinstance(atts[key], (NoneType)):
377  pass
378  else:
379  objAttribs.append(key)
380  except Exception:
381  pass
382 
383  if isinstance(obj, TypeType):
384  elName = obj.__name__
385  elif isinstance(obj, LmAttObj):
386  elName = obj.__name__
387  else:
388  elName = obj.__class__.__name__
389 
390  if isinstance(obj, Element):
391  if parent is None:
392  el = obj
393  else:
394  parent.append(obj)
395  el = obj
396  else:
397  if parent is None:
398  el = Element(elName, value=value, attrib=attrib)
399  else:
400  el = SubElement(parent, elName, value=value, attrib=attrib)
401 
402  for att in objAttribs:
403  subObj = getattr(obj, att)
404  if isinstance(subObj, ListType):
405  for i in subObj:
406  serialize(i, el)
407  elif isinstance(subObj, (MethodType, FunctionType, LambdaType,
408  BuiltinMethodType, BuiltinFunctionType)):
409  pass
410  elif isinstance(subObj, (StringType, IntType, FloatType, UnicodeType)):
411  SubElement(el, att, value=subObj)
412  elif isinstance(subObj, DictType):
413  sEl = SubElement(el, att)
414  for key in subObj.keys():
415  if isinstance(subObj[key], (StringType, IntType, FloatType, UnicodeType)):
416  SubElement(sEl, key, value=subObj[key])
417  elif isinstance(subObj[key], NoneType):
418  pass
419  else:
420  serialize(subObj[key], sEl)
