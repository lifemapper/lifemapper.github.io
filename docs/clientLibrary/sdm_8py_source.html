445  #
446  # @summary: Post a new Lifemapper experiment
447  # @param algorithm: An Lifemapper SDM algorithm object
448  # @param mdlScn: The id of the model scenario to use for the experiment
449  # [integer]
450  # @param occSetId: The id of the occurrence set to be used for the
451  # experiment. [integer]
452  # @param prjScns: (optional) List of projection scenario ids to use for the
453  # experiment [list of integers]
454  # @param mdlMask: (optional) A layer id mask to use when looking at the
455  # input climate layers of the model [integer]
456  # @param prjMask: (optional) A layer id mask to use when projecting onto a
457  # new set of climate layers [integer]
458  # @param email: (optional) Lifemapper will send a notification email to
459  # this address when the experiment has completed
460  # @param name: (optional) A name for this experiment
461  # @param description: (optional) A description for this experiment
462  # @return: Experiment
463  #
464  def postExperiment(self, algorithm, mdlScn, occSetId, prjScns=[],
465  mdlMask=None, prjMask=None,
466  email=None, name=None, description=None):
467  try:
468  algorithmParameters = algorithm.parameters
469  algoCode = algorithm.code
470  except:
471  algoCode = algorithm
472  algorithmParameters = []
473 
474  # Validate algorithm parameters
475  self.validateAlgorithmParameters(algorithmParameters, prjScns)
476 
477  algoParams = """\
478  <lm:parameters>
479  {params}
480  </lm:parameters>
481 """.format(params='\n '.join(
482  ["<lm:{name}>{value}</lm:{name}>".format(
483  name=param.name, value=param.value) \
484  for param in algorithmParameters])) \
485  if len(algorithmParameters) > 0 else ""
486  mMask = " <lm:modelMask>%s</lm:modelMask>" % mdlMask if mdlMask is not None else ""
487  pMask = " <lm:projectionMask>%s</lm:projectionMask>" % prjMask if prjMask is not None else ""
488  emailSection = " <lm:email>%s</lm:email>" % email if email is not None else ""
489  nameSection = " <lm:name>%s</lm:name>" % name if name is not None else ""
490  descSection = " <lm:description>%s</lm:description>" % description if description is not None else ""
491  prjSection = '\n'.join(([
492  " <lm:projectionScenario>{scnId}</lm:projectionScenario>".format(scnId=scnId) for scnId in prjScns]))
493  postXml = """\
494  <lm:request xmlns:lm="http://lifemapper.org"
495  xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
496  xsi:schemaLocation="http://lifemapper.org
497  /schemas/serviceRequest.xsd">
498  <lm:experiment>
499  <lm:algorithm>
500  <lm:algorithmCode>{algorithmCode}</lm:algorithmCode>
501 {algoParams}
502  </lm:algorithm>
503  <lm:occurrenceSetId>{occSetId}</lm:occurrenceSetId>
504  <lm:modelScenario>{mdlScn}</lm:modelScenario>
505 {mMask}
506 {email}
507 {name}
508 {description}
509 {projections}
510 {pMask}
511  </lm:experiment>
512  </lm:request>""".format(algorithmCode=algoCode,
513  algoParams=algoParams, occSetId=occSetId,
514  mdlScn=mdlScn, mMask=mMask, email=emailSection,
515  projections=prjSection, pMask=pMask,
516  name=nameSection, description=descSection)
517  url = "%s/services/sdm/experiments/" % self.cl.server
518  obj = self.cl.makeRequest(url,
519  method="POST",
520  body=postXml,
521  headers={"Content-Type": "application/xml"},
522  objectify=True).experiment
