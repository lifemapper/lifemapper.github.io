139 # ...............................................
140  def _assembleFilterString(self, filterString=None):
141  if filterString is not None:
142  for replaceStr, withStr in URL_ESCAPES:
143  filterString = filterString.replace(replaceStr, withStr)
144  else:
145  allFilters = self._otherFilters.copy()
146  if self._qFilters:
147  qVal = self._assembleQVal(self._qFilters)
148  allFilters[self._qKey] = qVal
149  filterString = self._assembleKeyValFilters(allFilters)
150  return filterString

152 # ...............................................
153  def _assembleKeyValFilters(self, ofDict):
154  for k, v in ofDict.iteritems():
155  if isinstance(v, BooleanType):
156  v = str(v).lower()
157  ofDict[k] = unicode(v).encode('utf-8')
158  filterString = urllib.urlencode(ofDict)
159  return filterString

161 # ...............................................
162  def _interpretQClause(self, key, val):
163  cls = None
164  if isinstance(val, (FloatType, IntType, StringType, UnicodeType)):
165  cls = '{}:{}'.format(key, str(val))
166  # Tuple for negated or range value
167  elif isinstance(val, TupleType):
168  # negated filter
169  if isinstance(val[0], BooleanType) and val[0] is False:
170  cls = 'NOT ' + key + ':' + str(val[1])
171  # range filter (better be numbers)
172  elif ((isinstance(val[0], IntType) or isinstance(val[0], FloatType))
173  and (isinstance(val[1], IntType) or isinstance(val[1], FloatType))):
174  cls = '{}:[{} TO {}]'.format(key, str(val[0]), str(val[1]))
175  else:
176  print 'Unexpected value type {}'.format(val)
177  else:
178  print 'Unexpected value type {}'.format(val)
179  return cls
180 
181 # ...............................................
182  def _assembleQItem(self, key, val):
183  itmClauses = []
184  # List for multiple values of same key
185  if isinstance(val, ListType):
186  for v in val:
187  itmClauses.append(self._interpretQClause(key, v))
188  else:
189  itmClauses.append(self._interpretQClause(key, val))
190  return itmClauses
191 
192 # ...............................................
193  def _assembleQVal(self, qDict):
194  clauses = []
195  qval = ''
196  # interpret dictionary
197  for key, val in qDict.iteritems():
198  clauses.extend(self._assembleQItem(key, val))
199  # convert to string
200  firstClause = ''
201  for cls in clauses:
202  if not firstClause and not cls.startswith('NOT'):
203  firstClause = cls
204  elif cls.startswith('NOT'):
205  qval = ' '.join((qval, cls))
206  else:
207  qval = ' AND '.join((qval, cls))
208  qval = firstClause + qval
209  return qval

254  def queryByGet(self, outputType='json'):
255  data = retcode = None
256  try:
257  response = requests.get(self.url, headers=self.headers)
258  except Exception, e:
259  try:
260  retcode = response.status_code
261  reason = response.reason
262  except:
263  reason = 'Unknown Error'
264  raise Exception('Failed on URL {}, code = {}, reason = {} ({})'
265  .format(self.url, retcode, reason, str(e)))
266 
267  try:
268  if outputType == 'json':
269  self.output = response.json()
270  else:
271  self.output = response.content
272  except Exception, e:
273  raise Exception('Failed to interpret output of URL {} ({})'
274  .format(self.url, str(e)))

276 # ...............................................
277  def queryByPost(self, outputType='json'):
278  allParams = self._otherFilters.copy()
279  allParams[self._qKey] = self._qFilters
280  queryAsString = json.dumps(allParams)
281  try:
282  response = requests.post(self.baseurl,
283  data=queryAsString,
284  headers=self.headers)
285  except Exception, e:
286  try:
287  retcode = response.status_code
288  reason = response.reason
289  except:
290  retcode = HTTPStatus.INTERNAL_SERVER_ERROR
291  reason = 'Unknown Error'
292  raise Exception('Failed on URL {}, code = {}, reason = {} ({})'.format(
293  self.url, retcode, reason, str(e)))
294 
295  if response.ok:
296  try:
297  if outputType == 'json':
298  self.output = response.json()
299  elif outputType == 'xml':
300  output = response.text
301  self.output = ET.fromstring(output)
302  else:
303  print 'Unrecognized output type {}'.format(outputType)
304  self.output = None
305  except Exception, e:
306  raise Exception('Failed to interpret output of URL {}, content = {}; ({})'
307  .format(self.url, response.content, str(e)))
308  else:
309  print('Failed query: {}'.format(response.content))
310  self.output = None
