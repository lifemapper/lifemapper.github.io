248  if os.path.isdir(filePath):
249  with zipfile.ZipFile(StringIO.StringIO(cnt), 'r', allowZip64=True) as zf:
250  # Check to see if files exist
251  nameList = zf.namelist()
252  if not overwrite:
253  for name in nameList:
254  if os.path.exists(os.path.join(filePath, name)):
255  raise Exception(
256  "File %s, already exists and overwrite is: %s" % (

258  zf.extractall(filePath)
259  else:
260  base, ext = os.path.splitext(filePath)
261  if ext == '.zip':
262  with open(filePath, 'wb') as outF:
263  outF.write(cnt)
264  elif ext == '.shp':
265  # Check to see if filePath exists
266  # Able to write if path doesn't exist or overwrite is true
267  if not os.path.exists(filePath) or overwrite:
268  with zipfile.ZipFile(StringIO.StringIO(cnt), 'r', allowZip64=True) as zf:
269  for name in zf.namelist():
270  fCnt = zf.read(name)
271  fBase, fExt = os.path.splitext(name)
272  with open(os.path.join(filePath, '%s%s' % (base, fExt)), 'wb') as outF:
273  outF.write(fCnt)
274 
275  else:
276  raise Exception, "%s already exists and overwrite is: %s" % (
