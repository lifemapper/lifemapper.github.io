34 # attached to XML elements and the object members would be the
35 # sub-elements of the element.
36 # @note: <someElement att1="value1" att2="value2">
37 # <subEl1>1</subEl1>
38 # <subEl2>banana</subEl2>
39 # </someElement>
40 #
41 # translates to:
42 #
43 # obj.subEl1 = 1
44 # obj.subEl2 = 'banana'
45 # obj.getAttributes() = {'att1': 'value1', 'att2': 'value2'}
