157  # @summary: Adds all of the tests found in a test configuration file
158  # @param filepath: The path to the file to find tests in
159  #
160  def addTestsFromFile(self, filepath, testFactory=None):
161  try:
162  if testFactory is None:
163  testFactory = LMTestFactory(self.testBuilders, **self.kwargs)
164 
165  if os.path.exists(filepath) and os.path.isfile(filepath):
166  with open(filepath) as inF:
167  testStr = inF.read()
168  testObj = deserialize(fromstring(testStr))
169  self.tests.extend(testFactory.getTests(testObj))
170  else:
171  raise Exception, "%s does not exist or is not a file" % filepath
172  except Exception, e:
173  print "Could not add tests from: %s" % filepath
174  print str(e)

237  def runTests(self):
238  try:
239  for t in self.tests:
240  # Check test level
241  if t.testLevel <= self.testLevel:
242  outputLog = self.getOutputLogger(t.name)
243  errorLog = self.getErrorLogger(t.name)
244 
245  t.setLogger(outputLog)
246  t.setErrorLogger(errorLog)
247 
248  startTime = self.getTime()
249  t.runTest()
250  endTime = self.getTime()
251 
252  t.evaluateTest()
253 
254  t.cleanup()
255 
256  # Success
257  if t.status == 0:
258  self.successful.append((t.name, startTime, endTime))
259  # Pedantic failure
260  elif t.status == 1 and not self.pedantic:
261  self.warnings.append((t.name, t.message))
262  # Failure
263  else:
264  self.failed.append((t.name, t.message))
265 
266  # Quick stop, on error or pedantic error
267  if self.quickStop and (
268  t.status == 2 or (self.pedantic and t.status == 1)):
269  break
270  else:
271  self.skipped.append(t.name)
272  except Exception, e:
273  print "An exception occurred, aborting"
274  print str(e)
275  sys.exit(2)
